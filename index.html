    function generator(){
        // Add your own words to the wordlist. Be carefull to obey the showed syntax

        var wordlist1 = ["Eagle", "Raven", "Tiger", "Great White Shark", "Orca", "Grizzly Bear", "Moose", "Cheetah", "Black Mamba", "Coyote", "Rooster", "Jelly Fish", "Scorpion", "Sting Ray", "Armadillo", "Sea Otter", "Beaver", "Viper", "Buffalo", "Hyena", "Narwhal", "Lemur", "Camel", "Fox", "Crocodile", "Rattlesnake", "Tortoise", "Giraffe", "Walrus", "Octopus", "Platypus", "Pangolin", "Meerkat", "Chameleon", "Falcon", "Tasmanian Devil", "Gorilla", "Chimpanze", "Hammerhead Shark", "Panda Bear", "Penguin", "Cougar", "Hippo", "Stallion", "Porcupine", "Bull", "Mongoose", "Rabbit/Hare", "Seal", "Elephant", "Tarantula", "Hawk", "Peacock", "Sloth", "Dolphin", "Anaconda", "Kangaroo", "Lion"];

        // Random numbers are made 
        var randomNumber1 = parseInt(Math.random() * wordlist1.length);
        var name = wordlist1[randomNumber1];            

        //alert(name); //Remove first to slashes to alert the name

        //If there's already a name it is removed  
        if(document.getElementById("result")){
            document.getElementById("placeholder").removeChild(document.getElementById("result"));
        }
        // A div element is created to show the generated name. The Name is added as a textnode. Textnode is added to the placeholder.
        var element = document.createElement("div");
        element.setAttribute("id", "result");
        element.appendChild(document.createTextNode(name));
        document.getElementById("placeholder").appendChild(element);
    }       
</script>
